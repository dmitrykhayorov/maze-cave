106template <typename T>
107std::vector<point> Solver<T>::DefineAccessableNeighbors(point p,
108 const maze_t &maze) {
109 std::vector<point> neighbors;
110 // check north
111 if (p.second > 0 && !maze.mazeBottomBorder[p.second - 1][p.first]) {
112 neighbors.push_back(point{p.first, p.second - 1});
113 }
114 // check south
115 if (p.second < maze.cols - 1 && !maze.mazeBottomBorder[p.second][p.first]) {
116 neighbors.push_back(point{p.first, p.second + 1});
117 }
118 // check west
119 if (p.first > 0 && !maze.mazeRightBorder[p.second][p.first - 1]) {
120 neighbors.push_back(point{p.first - 1, p.second});
121 }
122 // check east
123 if (p.first < maze.rows - 1 && !maze.mazeRightBorder[p.second][p.first]) {
124 neighbors.push_back(point{p.first + 1, p.second});
125 }
126 return neighbors;

128template <typename T>
129std::vector<point> Solver<T>::DefineAccessableNeighbors(point p,
130 const cave_t &cave) {
131 std::vector<point> neighbors;
132 int x = p.first, y = p.second;
133 if (cave.cave.empty()) return neighbors;
134 if (y >= cave.cols || y < 0) return neighbors;
135 if (x >= cave.rows || x < 0) return neighbors;
136 if (cave.cave[y][x]) {
137 return neighbors;
138 }
139
140 for (int i = -1; i <= 1; ++i) {
141 if (y + i < 0 || y + i >= cave.cols) continue;
142 for (int j = -1; j <= 1; j++) {
143 if (i == 0 && j == 0) continue;
144 if (x + j < 0 || x + j >= cave.rows) continue;
145 int sum = i + j;
146 if (sum == -2 || sum == 0 || sum == 2) continue;
147 if (!cave.cave[y + i][x + j]) {
148 neighbors.push_back(point{x + j, y + i});
149 }
150 }
151 }
152
153 return neighbors;
